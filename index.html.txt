            const tasksContainer = document.getElementById('tasks-container');
            const phaseFilters = document.getElementById('phase-filters');
            let currentPhaseFilter = 'all';
            let progressChart;

            function renderTasks() {
                tasksContainer.innerHTML = '';
                const filteredTasks = tasksData.filter(task => 
                    currentPhaseFilter === 'all' || task.phase == currentPhaseFilter
                );

                if (filteredTasks.length === 0) {
                    tasksContainer.innerHTML = `<p class="text-center text-stone-500 col-span-full">لا توجد مهام في هذه المرحلة.</p>`;
                    return;
                }

                filteredTasks.forEach(task => {
                    const card = document.createElement('div');
                    card.className = `task-card status-${task.status} p-4 rounded-lg shadow-md bg-white`;
                    card.dataset.taskId = task.id;
                    
                    card.innerHTML = `
                        <div class="flex justify-between items-start">
                            <h3 class="text-lg font-bold text-stone-800">${task.title}</h3>
                            <div class="flex items-center">
                                <input type="checkbox" data-task-id="${task.id}" class="task-status-checkbox h-5 w-5 text-teal-600 border-gray-300 rounded focus:ring-teal-500" ${task.status === 'done' ? 'checked' : ''}>
                            </div>
                        </div>
                        <div class="text-sm text-stone-500 mt-2">
                            <p>👤 <span class="font-semibold">${task.responsible}</span></p>
                            <p>📅 <span class="font-semibold">${task.deadline}</span></p>
                        </div>
                        <button class="details-btn text-sm text-teal-600 font-semibold mt-4">عرض التفاصيل</button>
                        <div class="task-card-details mt-4 pt-4 border-t border-gray-200">
                            <h4 class="font-bold text-stone-700">التفاصيل الدقيقة:</h4>
                            <p class="text-sm text-stone-600 mt-1">${task.details}</p>
                            <h4 class="font-bold text-stone-700 mt-3">توصية القائد:</h4>
                            <p class="text-sm text-amber-800 bg-amber-50 p-2 rounded-md mt-1 italic">"${task.tip}"</p>
                        </div>
                    `;
                    tasksContainer.appendChild(card);
                });
            }

            function updateProgress() {
                const totalTasks = tasksData.length;
                const completedTasks = tasksData.filter(t => t.status === 'done').length;
                const progressPercentage = totalTasks > 0 ? Math.round((completedTasks / totalTasks) * 100) : 0;

                document.getElementById('progressText').textContent = `${progressPercentage}%`;
                document.getElementById('tasksCompletedText').textContent = `${completedTasks} من ${totalTasks} مهمة تم إنجازها`;

                progressChart.data.datasets[0].data = [completedTasks, totalTasks - completedTasks];
                progressChart.update();
            }

            function setupChart() {
                const ctx = document.getElementById('progressChart').getContext('2d');
                progressChart = new Chart(ctx, {
                    type: 'doughnut',
                    data: {
                        labels: ['أنجزت', 'متبقية'],
                        datasets: [{
                            data: [0, tasksData.length],
                            backgroundColor: ['#14b8a6', '#e5e7eb'],
                            borderColor: ['#ffffff'],
                            borderWidth: 4,
                            hoverOffset: 4
                        }]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        cutout: '70%',
                        plugins: {
                            legend: {
                                display: false
                            },
                            tooltip: {
                                enabled: true
                            }
                        }
                    }
                });
            }

            phaseFilters.addEventListener('click', (e) => {
                if (e.target.tagName === 'BUTTON') {
                    currentPhaseFilter = e.target.dataset.phase;
                    document.querySelectorAll('.phase-btn').forEach(btn => {
                        btn.classList.remove('bg-teal-600', 'text-white');
                        btn.classList.add('bg-white', 'text-stone-700');
                    });
                    e.target.classList.add('bg-teal-600', 'text-white');
                    e.target.classList.remove('bg-white', 'text-stone-700');
                    renderTasks();
                }
            });

            tasksContainer.addEventListener('click', (e) => {
                const taskId = e.target.dataset.taskId;
                if (e.target.classList.contains('task-status-checkbox')) {
                    const task = tasksData.find(t => t.id == taskId);
                    task.status = e.target.checked ? 'done' : 'not-started';
                    e.target.closest('.task-card').className = `task-card status-${task.status} p-4 rounded-lg shadow-md bg-white`;
                    updateProgress();
                }

                if (e.target.classList.contains('details-btn')) {
                    const detailsEl = e.target.nextElementSibling;
                    detailsEl.classList.toggle('open');
                    e.target.textContent = detailsEl.classList.contains('open') ? 'إخفاء التفاصيل' : 'عرض التفاصيل';
                }
            });

            setupChart();
            renderTasks();
            updateProgress();
        });
    </script>
